40 inline std::vector<HullPoint> computeJarvisHull(const std::vector<HullPoint>& point_set) {
41 size_t n = point_set.size();
42 if (n < 3) return {};
43
44 // Find leftmost point
45 size_t leftmost_pos = 0;
46 for (size_t i = 1; i < n; ++i) {
47 if (point_set[i].x < point_set[leftmost_pos].x ||
48 (point_set[i].x == point_set[leftmost_pos].x &&
49 point_set[i].y < point_set[leftmost_pos].y)) {
50 leftmost_pos = i;
51 }
52 }

54 std::vector<HullPoint> hull_points;
55 size_t current_pos = leftmost_pos, next_pos;
56
57 do {
58 hull_points.push_back(point_set[current_pos]);
59 next_pos = (current_pos + 1) % n;
60
61 for (size_t i = 0; i < n; ++i) {
62 if (checkOrientation(point_set[current_pos], point_set[i], point_set[next_pos]) == 2) {
63 next_pos = i;

65 }

67 current_pos = next_pos;
68 } while (current_pos != leftmost_pos);
